</br>N = 8 -> "8, 7, 6, 5, 4, 3, 2, 1"

</br>M = 1; N = 15 -> 120
</br>M=4; N = 8. -> 30

</br>m = 2, n = 3 -> A(m,n) = 9
</br>m = 3, n = 2 -> A(m,n) = 29
